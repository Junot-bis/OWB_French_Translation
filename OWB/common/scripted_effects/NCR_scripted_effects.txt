setup_NCR_enclave = {
	# add Reno enclave to this one
	if = {
		limit = {
			NEW = {
				has_cosmetic_tag = ENC
			}
		}
		if = {
			limit = {
				NEW = {
					is_ai = yes
					is_in_faction = no
				}
			}
			every_other_country = {
				limit = {
					is_puppet_of = NEW
				}
				NEW = { end_puppet = PREV }
				ENC = { puppet = PREV }
			}
			NEW = {
				every_unit_leader = {
					set_nationality = ENC
				}
			}
			annex_country = { target = NEW transfer_troops = yes }
		}
		else_if = {
			limit = {
				NEW = {
					is_ai = yes
					is_in_faction = yes
					is_faction_leader = no
				}
			}
			create_faction = western_enclave
			remove_from_faction = NEW
			add_to_faction = NEW
			add_ai_strategy = {
				type = alliance
				id = "NEW"
				value = 200
			}
			NEW = {
				add_ai_strategy = {
					type = alliance
					id = "ENC"
					value = 200
				}
			}
			every_other_country = {
				limit = {
					is_puppet_of = NEW
				}
				remove_from_faction = PREV
				ENC = {
					add_to_faction = PREV
					add_ai_strategy = {
						type = alliance
						id = "PREV"
						value = 200
					}
				}
				PREV = {
					add_ai_strategy = {
						type = alliance
						id = "ENC"
						value = 200
					}
				}
			}
		}
		else_if = {
			limit = {
				NEW = {
					is_ai = yes
					is_in_faction = yes
					is_faction_leader = yes
				}
			}
			create_faction = western_enclave
			remove_from_faction = NEW
			add_to_faction = NEW
			add_ai_strategy = {
				type = alliance
				id = "NEW"
				value = 200
			}
			NEW = {
				add_ai_strategy = {
					type = alliance
					id = "ENC"
					value = 200
				}
			}
			every_other_country = {
				limit = {
					is_puppet_of = NEW
				}
				remove_from_faction = PREV
				ENC = {
					add_to_faction = PREV
					add_ai_strategy = {
						type = alliance
						id = "PREV"
						value = 200
					}
				}
				PREV = {
					add_ai_strategy = {
						type = alliance
						id = "ENC"
						value = 200
					}
				}
			}
		}
	}
	set_politics = {
		parties = {
			people = { popularity = 0 }
			elites = { popularity = 80 }
			intellectuals = { popularity = 20 }
			ruler = { popularity = 0 }
		}
		ruling_party = elites
		elections_allowed = no
	}
	hidden_effect = {
		set_technology = {
			power_armour_enclave1 = 1
			tesla_weaponry = 1
			vertiberd_gunship = 1
			vertiberd_transport1 = 1
		}
		load_oob = enclavecivilwar
		retire_country_leader = yes
		create_country_leader = {
			name = "Grant C. Hayes"
			desc = "Hayes_DESC"
			picture = "gfx/leaders/ENC/Portrait_Hayes.dds"
			expire = "3000.1.1"
			ideology = militarist_technocracy

			traits = {

			}
		}
	}
	if = {
		limit = {
			NCR = {
				has_completed_focus = ncr_integration
			}
			BJS = {
				exists = yes
			}
		}
		NCR = {
			release_puppet = BJS
		}
		ENC = {
			puppet = BJS
		}
	}
}
